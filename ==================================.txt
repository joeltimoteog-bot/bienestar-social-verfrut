// ===== HELPER: HASH SHA-256 =====
function bs_hashPassword(password) {
  const bytes = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, password, Utilities.Charset.UTF_8);
  return bytes.map(function(b) { return ('0' + (b & 0xFF).toString(16)).slice(-2); }).join('');
}

function bs_crearQueja(usuario, datos) {

    if (!sheet) return { ok: false, message: 'Hoja BS_Quejas no existe' };

    const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]
                      .map(function(h){ return String(h).trim(); });
    const lastRow = sheet.getLastRow();
    const id = lastRow > 1 ? (parseInt(sheet.getRange(lastRow, 1).getValue()) || 0) + 1 : 1;
    const now = new Date();

    const v = {
      id: id,

      clasificacion: datos.clasificacion || '',
      canal: datos.canal || '',

      detalle: datos.detalle || '',

      accion_mejora: datos.accion_mejora || '',
      coordinacion: datos.coordinacion || '',
      fecha_levantamiento_ac: datos.fecha_levantamiento_ac ? new Date(datos.fecha_levantamiento_ac) : '',
      frecuencia_seguimiento_ac: datos.frecuencia_seguimiento_ac || '',
      fecha_seguimiento_am: datos.fecha_seguimiento_am ? new Date(datos.fecha_seguimiento_am) : '',
      observaciones: datos.observaciones || '',
      resultado: datos.resultado || 'En proceso',
      documentos: datos.documentos || '',
      empresa: datos.empresa || '',
      created_by: usuario,
      created_at: now,
      updated_at: now,

      cantidad_quejas: parseInt(datos.cantidad_quejas) || 0,
     buzones_detalle: datos.buzones_detalle || '',

function bs_listarQuejas(filtros) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName(BS_CONFIG.HOJA_QUEJAS);
    if (!sheet) return { ok: false, message: 'Hoja BS_Quejas no existe' };
    const data = sheet.getDataRange().getValues();
    if (data.length <= 1) return { ok: true, items: [], total: 0 };
    const headers = data[0];
    const rows = data.slice(1);
    let items = rows.map(function(row) {
      const obj = {};
      headers.forEach(function(h, i) {
        let v = row[i];
        if (v instanceof Date) v = v.toISOString();
        obj[h] = v;
      });
      return obj;
    });
    if (filtros) {
      if (filtros.empresa) items = items.filter(function(i) { return i.empresa === filtros.empresa; });
      if (filtros.clasificacion) items = items.filter(function(i) { return i.clasificacion === filtros.clasificacion; });
      if (filtros.resultado) items = items.filter(function(i) { return i.resultado === filtros.resultado; });
    }
    items.sort(function(a, b) { return (b.id || 0) - (a.id || 0); });
    return { ok: true, items: items, total: items.length };
  } catch (err) {
    return { ok: false, message: 'Error: ' + err.message };
  }
}

function bs_actualizarQueja(usuario, id, datos) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName(BS_CONFIG.HOJA_QUEJAS);
    if (!sheet) return { ok: false, message: 'Hoja BS_Quejas no existe' };
    const data = sheet.getDataRange().getValues();
    const headers = data[0];
    for (let i = 1; i < data.length; i++) {
      if (parseInt(data[i][0]) === parseInt(id)) {
        const rowNum = i + 1;
        const updatable = ['fecha_recepcion','clasificacion','canal','categoria','sector','detalle','accion_correctiva','accion_mejora','coordinacion','fecha_levantamiento_ac','frecuencia_seguimiento_ac','fecha_seguimiento_am','observaciones','resultado','documentos','empresa','cantidad_buzones','cantidad_quejas','buzones_detalle','tipo_lugar','seguimiento','seguimiento_frecuencia','seguimiento_inicio','seguimiento_fin','seguimiento_dias','seguimiento_alerta'];
        updatable.forEach(function(field) {
          if (datos[field] !== undefined) {
            const colIdx = headers.indexOf(field);
            if (colIdx >= 0) {
              let val = datos[field];
              if (field.indexOf('fecha_') === 0 && val) val = new Date(val);
              sheet.getRange(rowNum, colIdx + 1).setValue(val);
            }
          }
        });
        const colUpdated = headers.indexOf('updated_at');
        if (colUpdated >= 0) sheet.getRange(rowNum, colUpdated + 1).setValue(new Date());
        return { ok: true, message: 'Queja actualizada' };
      }
    }
    return { ok: false, message: 'Queja no encontrada' };
  } catch (err) {
    return { ok: false, message: 'Error: ' + err.message };
  }
}

function bs_actualizarSubsidio(usuario, id, datos) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName(BS_CONFIG.HOJA_SUBSIDIOS);
    if (!sheet) return { ok: false, message: 'Hoja BS_Subsidios no existe' };
    const data = sheet.getDataRange().getValues();
    const headers = data[0];
    const updatables = ['anio','mes','tipo','numero','apellido_paterno','apellido_materno','nombres','descripcion','estado','dias','calculo_por_dia','pago_empresa','reembolso_por_mes','recuperado_essalud','monto_total','periodo','expediente','inicio','termino','observaciones','diferencias','tipo_pago_planilla','estado_general','empresa'];
    const dateFields = ['inicio','termino'];
    const numFields = ['dias','calculo_por_dia','pago_empresa','reembolso_por_mes','recuperado_essalud','monto_total','diferencias','anio'];
    for (let i = 1; i < data.length; i++) {
      if (parseInt(data[i][0]) === parseInt(id)) {
        const rowNum = i + 1;
        updatables.forEach(function(field) {
          if (datos[field] !== undefined) {
            const colIdx = headers.indexOf(field);
            if (colIdx >= 0) {
              let val = datos[field];
              if (dateFields.indexOf(field) >= 0 && val) val = new Date(val);
              else if (numFields.indexOf(field) >= 0) val = parseFloat(val) || 0;
              sheet.getRange(rowNum, colIdx + 1).setValue(val);
            }
          }
        });
        const colUpd = headers.indexOf('updated_at');
        if (colUpd >= 0) sheet.getRange(rowNum, colUpd + 1).setValue(new Date());
        return { ok: true, message: 'Subsidio actualizado' };
      }
    }
    return { ok: false, message: 'Subsidio no encontrado' };
  } catch (err) { return { ok: false, message: err.message }; }
}

function getEstadisticasBienestar(permisos) {
  try {
    permisos = permisos || {};
    var verHost = permisos.hostigamiento === true || permisos.mod_hostigamiento === true;
    var ss = SpreadsheetApp.getActiveSpreadsheet();

    function leer(nombre){
      var sh = ss.getSheetByName(nombre);
      if(!sh) return {h:[], rows:[]};
      var lr = sh.getLastRow(), lc = sh.getLastColumn();
      if(lc===0) return {h:[], rows:[]};
      var data = sh.getRange(1,1,Math.max(1,lr),lc).getValues();
      var h = data.shift();
      return {h:h, rows: lr<2 ? [] : data};
    }
    function ix(h,n){ return h.indexOf(n); }
    function txt(v){ return String(v==null?'':v).trim(); }
    function num(v){ var x=Number(txt(v).replace(/,/g,'')); return isNaN(x)?0:x; }
    function ym(v){
      if(v instanceof Date) return v.getFullYear()+'-'+('0'+(v.getMonth()+1)).slice(-2);
      var s=txt(v); var m=s.match(/(\d{4})\D(\d{1,2})/);
      return m ? m[1]+'-'+('0'+m[2]).slice(-2) : '';
    }
    function emp(v){ var s=txt(v).toUpperCase(); return s.indexOf('RAPEL')>=0?'RAPEL':(s.indexOf('VERFRUT')>=0?'VERFRUT':'OTRO'); }
    function toDate(v){ if(v instanceof Date) return v; var s=txt(v); if(!s) return null; var d=new Date(s); return isNaN(d.getTime())?null:d; }
    function fcorta(d){ return d ? (d.getFullYear()+'-'+('0'+(d.getMonth()+1)).slice(-2)+'-'+('0'+d.getDate()).slice(-2)) : ''; }

    var tend = {};
    function bump(periodo, mod){ if(!periodo) return; if(!tend[periodo]) tend[periodo]={quejas:0,accidentes:0,hostigamiento:0,subsidios:0}; tend[periodo][mod]++; }

    var recientes = [];
    function addReciente(modulo, icon, fecha, titulo, estado, empresa){
      var d = toDate(fecha);
      recientes.push({ modulo:modulo, icon:icon, fecha:fcorta(d), titulo:titulo, estado:estado, empresa:empresa, _ts: d?d.getTime():0 });
    }

    // ---- QUEJAS ----
    var q = leer('BS_Quejas');
    var qRes=ix(q.h,'resultado'), qFec=ix(q.h,'fecha_recepcion'), qEmp=ix(q.h,'empresa'), qCat=ix(q.h,'categoria');
    var quejas={total:0,resueltas:0,pendientes:0,porEmpresa:{RAPEL:0,VERFRUT:0}};
    q.rows.forEach(function(r){
      quejas.total++;
      if(/resuelt|cerrad/i.test(txt(r[qRes]))) quejas.resueltas++;
      if(qEmp>=0 && quejas.porEmpresa[emp(r[qEmp])]!==undefined) quejas.porEmpresa[emp(r[qEmp])]++;
      bump(ym(r[qFec]),'quejas');
      addReciente('Quejas','📋', r[qFec], 'Queja'+(qCat>=0&&txt(r[qCat])?' · '+txt(r[qCat]):''), txt(r[qRes])||'Pendiente', qEmp>=0?emp(r[qEmp]):'');
    });
    quejas.pendientes = quejas.total - quejas.resueltas;

    // ---- ACCIDENTES ----
    var a = leer('BS_Accidentes_Casos');
    var aEst=ix(a.h,'estado_caso'), aFec=ix(a.h,'dia_reportado'), aFec2=ix(a.h,'dia_accidente'), aEmp=ix(a.h,'empresa'), aParte=ix(a.h,'parte_cuerpo'), aTipo=ix(a.h,'tipo_accidente');
    var accidentes={total:0,cerrados:0,enSeguimiento:0,porEmpresa:{RAPEL:0,VERFRUT:0}};
    a.rows.forEach(function(r){
      accidentes.total++;
      if(/cerrad/i.test(txt(r[aEst]))) accidentes.cerrados++;
      if(aEmp>=0 && accidentes.porEmpresa[emp(r[aEmp])]!==undefined) accidentes.porEmpresa[emp(r[aEmp])]++;
      var fa = (aFec>=0 && txt(r[aFec])!=='') ? r[aFec] : (aFec2>=0?r[aFec2]:'');
      bump(ym(fa),'accidentes');
      var det = (aParte>=0&&txt(r[aParte])) ? txt(r[aParte]) : (aTipo>=0?txt(r[aTipo]):'');
      addReciente('Accidentes','🩺', fa, 'Accidente'+(det?' · '+det:''), txt(r[aEst])||'En seguimiento', aEmp>=0?emp(r[aEmp]):'');
    });
    accidentes.enSeguimiento = accidentes.total - accidentes.cerrados;

    // ---- HOSTIGAMIENTO (gateado, confidencial) ----
    var hostigamiento={acceso:verHost, total:0};
    if(verHost){
      var hh = leer('BS_Hostigamiento');
      var hFec=ix(hh.h,'fecha_denuncia'), hSt=ix(hh.h,'status');
      hh.rows.forEach(function(r){
        hostigamiento.total++;
        bump(ym(r[hFec]),'hostigamiento');
        addReciente('Hostigamiento','🔒', r[hFec], 'Denuncia registrada', hSt>=0?(txt(r[hSt])||'En proceso'):'En proceso', '');
      });
    }

    // ---- SUBSIDIOS ----
    var s = leer('BS_Subsidios');
    var sEst=ix(s.h,'estado_general'), sAnio=ix(s.h,'anio'), sMes=ix(s.h,'mes'),
        sIni=ix(s.h,'inicio'), sRec=ix(s.h,'recuperado_essalud'), sTot=ix(s.h,'monto_total'), sEmp=ix(s.h,'empresa'), sTipo=ix(s.h,'tipo');
    var subsidios={total:0,recuperadoEssalud:0,montoTotal:0,porEmpresa:{RAPEL:0,VERFRUT:0},porEstado:{}};
    s.rows.forEach(function(r){
      subsidios.total++;
      subsidios.recuperadoEssalud += num(r[sRec]);
      subsidios.montoTotal += num(r[sTot]);
      if(sEmp>=0 && subsidios.porEmpresa[emp(r[sEmp])]!==undefined) subsidios.porEmpresa[emp(r[sEmp])]++;
      var e2 = txt(r[sEst]) || 'Sin estado';
      subsidios.porEstado[e2] = (subsidios.porEstado[e2]||0)+1;
      var per = (sAnio>=0 && txt(r[sAnio])) ? ym(txt(r[sAnio])+'-'+txt(r[sMes])) : ym(r[sIni]);
      bump(per,'subsidios');
      var fsub = (sIni>=0 && txt(r[sIni])!=='') ? r[sIni] : ((sAnio>=0&&txt(r[sAnio])) ? (txt(r[sAnio])+'-'+('0'+txt(r[sMes])).slice(-2)+'-01') : '');
      addReciente('Subsidios','💰', fsub, 'Subsidio'+(sTipo>=0&&txt(r[sTipo])?' · '+txt(r[sTipo]):''), e2, sEmp>=0?emp(r[sEmp]):'');
    });

    // ---- TENDENCIA ----
    var tendencia = Object.keys(tend).sort().slice(-12).map(function(p){
      return {periodo:p, quejas:tend[p].quejas, accidentes:tend[p].accidentes, hostigamiento:tend[p].hostigamiento, subsidios:tend[p].subsidios};
    });

    // ---- RECIENTES (top 10 por fecha desc) ----
    recientes.sort(function(x,y){ return y._ts - x._ts; });
    recientes = recientes.slice(0,10).map(function(it){ return {modulo:it.modulo, icon:it.icon, fecha:it.fecha, titulo:it.titulo, estado:it.estado, empresa:it.empresa}; });

    return { ok:true, quejas:quejas, accidentes:accidentes, hostigamiento:hostigamiento, subsidios:subsidios, tendencia:tendencia, recientes:recientes, generado:new Date().toISOString() };
  } catch(err){
    return { ok:false, message:String(err) };
  }
}

function _aFecha(v){
  if(v instanceof Date) return new Date(v.getFullYear(), v.getMonth(), v.getDate());
  var s=String(v==null?'':v).trim(); if(!s) return null;
  var d=new Date(s); return isNaN(d.getTime())?null:new Date(d.getFullYear(),d.getMonth(),d.getDate());
}
function _ymd(d){ return d ? (d.getFullYear()+'-'+('0'+(d.getMonth()+1)).slice(-2)+'-'+('0'+d.getDate()).slice(-2)) : ''; }
function _esFeriado(d){ return ALERTAS_CONFIG.FERIADOS.indexOf(_ymd(d))>=0; }
function _esHabil(d){ var w=d.getDay(); return w!==0 && w!==6 && !_esFeriado(d); }
function _sumarHabiles(fecha, n){ var d=new Date(fecha.getTime()), c=0; while(c<n){ d.setDate(d.getDate()+1); if(_esHabil(d)) c++; } return d; }
function _sumarCalendario(fecha, n){ var d=new Date(fecha.getTime()); d.setDate(d.getDate()+n); return d; }
function _hoy(){ var n=new Date(); return new Date(n.getFullYear(),n.getMonth(),n.getDate()); }
function _diasEntre(a,b){ return Math.round((b.getTime()-a.getTime())/86400000); }
function _estadoVenc(fechaLimite){
  var dr=_diasEntre(_hoy(), fechaLimite);
  return { estado: dr<0 ? 'rojo' : (dr<=ALERTAS_CONFIG.UMBRAL_AMBAR_DIAS ? 'ambar' : 'verde'), diasRestantes:dr };
}
 
function calcularAlertasBienestar(permisos){
  try {
    permisos = permisos || {};
    var ver = {
      quejas:        permisos.quejas===true        || permisos.mod_quejas===true,
      accidentes:    permisos.accidentes===true    || permisos.mod_accidentes===true,
      hostigamiento: permisos.hostigamiento===true || permisos.mod_hostigamiento===true
    };
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    function leer(n){ var sh=ss.getSheetByName(n); if(!sh) return {h:[],rows:[]}; var lr=sh.getLastRow(),lc=sh.getLastColumn(); if(lc===0) return {h:[],rows:[]}; var data=sh.getRange(1,1,Math.max(1,lr),lc).getValues(); var h=data.shift(); return {h:h,rows:lr<2?[]:data}; }
    function ix(h,n){ return h.indexOf(n); }
    function txt(v){ return String(v==null?'':v).trim(); }
    function emp(v){ var s=txt(v).toUpperCase(); return s.indexOf('RAPEL')>=0?'RAPEL':(s.indexOf('VERFRUT')>=0?'VERFRUT':''); }
 
    var alertas=[];
    function push(modulo,id,regla,descripcion,fb,fl,empresa){
      var e=_estadoVenc(fl);
      alertas.push({ modulo:modulo, id:id, regla:regla, descripcion:descripcion, fechaBase:_ymd(fb), fechaLimite:_ymd(fl), diasRestantes:e.diasRestantes, estado:e.estado, empresa:empresa });
    }
 
    // ---- QUEJAS ----
    if(ver.quejas){
      var q=leer('BS_Quejas');
      var qId=ix(q.h,'id'), qFec=ix(q.h,'fecha_recepcion'), qRes=ix(q.h,'resultado'), qEmp=ix(q.h,'empresa'), qCat=ix(q.h,'categoria');
      q.rows.forEach(function(r){
        if(/resuelt|cerrad/i.test(txt(r[qRes]))) return;
        var fb=_aFecha(r[qFec]); if(!fb) return;
        var fl = ALERTAS_CONFIG.PLAZO_QUEJAS_TIPO==='calendario' ? _sumarCalendario(fb,ALERTAS_CONFIG.PLAZO_QUEJAS_DIAS) : _sumarHabiles(fb,ALERTAS_CONFIG.PLAZO_QUEJAS_DIAS);
        push('Quejas', txt(r[qId]), 'Atencion/resolucion', 'Queja'+(qCat>=0&&txt(r[qCat])?' · '+txt(r[qCat]):''), fb, fl, qEmp>=0?emp(r[qEmp]):'');
      });
    }
 
    // ---- HOSTIGAMIENTO (confidencial, sin nombres) ----
    if(ver.hostigamiento){
      var h=leer('BS_Hostigamiento');
      var hId=ix(h.h,'id'), hFec=ix(h.h,'fecha_denuncia'), hMed=ix(h.h,'medida_proteccion'), hSt=ix(h.h,'status');
      h.rows.forEach(function(r){
        var fb=_aFecha(r[hFec]); if(!fb) return;
        var cerrado=/cerrad|resuelt|archivad/i.test(txt(r[hSt]));
        if(hMed<0 || !txt(r[hMed])) push('Hostigamiento', txt(r[hId]), 'Medidas de proteccion (3 dh)', 'Caso de hostigamiento #'+txt(r[hId]), fb, _sumarHabiles(fb,3), '');
        if(!cerrado)               push('Hostigamiento', txt(r[hId]), 'Informe de investigacion (15 dcal)', 'Caso de hostigamiento #'+txt(r[hId]), fb, _sumarCalendario(fb,15), '');
      });
    }
 
    // ---- ACCIDENTES (solo mortal/grave -> 24h MTPE) ----
    if(ver.accidentes){
      var a=leer('BS_Accidentes_Casos');
      var aId=ix(a.h,'id'), aAcc=ix(a.h,'dia_accidente'), aTipo=ix(a.h,'tipo_accidente'), aEstA=ix(a.h,'estatus_accidente'), aEstC=ix(a.h,'estado_caso'), aEmp=ix(a.h,'empresa'), aParte=ix(a.h,'parte_cuerpo');
      a.rows.forEach(function(r){
        if(!/mortal|grave|incidente peligroso/i.test(txt(r[aTipo])+' '+txt(r[aEstA]))) return;
        if(/cerrad/i.test(txt(r[aEstC]))) return;
        var fb=_aFecha(r[aAcc]); if(!fb) return;
        push('Accidentes', txt(r[aId]), 'Notificacion MTPE (24h)', 'Accidente'+(aParte>=0&&txt(r[aParte])?' · '+txt(r[aParte]):'')+' (mortal/grave)', fb, _sumarCalendario(fb,1), aEmp>=0?emp(r[aEmp]):'');
      });
    }
 
    var orden={rojo:0,ambar:1,verde:2};
    alertas.sort(function(x,y){ return (orden[x.estado]-orden[y.estado]) || (x.diasRestantes-y.diasRestantes); });
    var resumen={rojo:0,ambar:0,verde:0};
    alertas.forEach(function(al){ resumen[al.estado]++; });
 
    return { ok:true, alertas:alertas, resumen:resumen, generado:new Date().toISOString() };
  } catch(err){ return { ok:false, message:String(err) }; }
}

var PADRON_CONFIG = {
  SPREADSHEET_ID: '1q79u2S3ZI_Qc-YnDzgnQwyv4jL7pxTuARiXICPPXgZw', // <-- confirma que aqui estan Trabajadores_RAPEL/VERFRUT

  COL_DNI: 0, COL_INICIO: 5, COL_TERMINO: 17, COL_EMPRESA: 13, COL_NOMBRE: 14, // A,F,R,N,O (0-based)
  PERIODO_PRUEBA_DIAS: 90,      // periodo de prueba (3 meses)
  INDETERMINADO_MESES: 54,      // 4 años 6 meses
  PROXIMO_INDET_DIAS: 30,       // aviso "proximo a indeterminado"
  POR_VENCER_DIAS: 30           // aviso "contrato por vencer"
};

function clasificarTrabajadores(){
  try{
    var C = PADRON_CONFIG;
    var ss = SpreadsheetApp.openById(C.SPREADSHEET_ID);
    var hoy = _hoy();
    function mesesEntre(ini, fin){
      var m = (fin.getFullYear()-ini.getFullYear())*12 + (fin.getMonth()-ini.getMonth());
      if (fin.getDate() < ini.getDate()) m--;
      return m;
    }
    var porTipo = {prueba:0, plazoFijo:0, indeterminado:0, sinFecha:0};
    var porEmpresa = { RAPEL:{prueba:0,plazoFijo:0,indeterminado:0}, VERFRUT:{prueba:0,plazoFijo:0,indeterminado:0} };
    var proximosIndet = [], porVencer = [], total = 0;

    C.HOJAS.forEach(function(nombreHoja){
      var sh = ss.getSheetByName(nombreHoja);
      if(!sh) return;
      var lr = sh.getLastRow(); if(lr<2) return;
      var data = sh.getRange(2,1,lr-1,sh.getLastColumn()).getValues(); // desde fila 2 (sin encabezado)
      var emp = nombreHoja.indexOf('RAPEL')>=0 ? 'RAPEL' : 'VERFRUT';
      data.forEach(function(r){
        var dni = String(r[C.COL_DNI]||'').trim(); if(!dni) return;
        total++;
        var ini = _aFecha(r[C.COL_INICIO]);
        if(!ini){ porTipo.sinFecha++; return; }
        var dias = Math.round((hoy.getTime()-ini.getTime())/86400000);
        var meses = mesesEntre(ini, hoy);
        var tipo = (dias <= C.PERIODO_PRUEBA_DIAS) ? 'prueba'
                 : (meses >= C.INDETERMINADO_MESES) ? 'indeterminado' : 'plazoFijo';
        porTipo[tipo]++;
        if(porEmpresa[emp]) porEmpresa[emp][tipo]++;
        var nombre = String(r[C.COL_NOMBRE]||'').trim();

        // proximo a indeterminado (le faltan <= X dias para los 54 meses)
        if(tipo==='plazoFijo'){
          var fIndet = new Date(ini.getFullYear(), ini.getMonth()+C.INDETERMINADO_MESES, ini.getDate());
          var dPara = Math.round((fIndet.getTime()-hoy.getTime())/86400000);
          if(dPara>=0 && dPara<=C.PROXIMO_INDET_DIAS) proximosIndet.push({dni:dni,nombre:nombre,empresa:emp,fecha:_ymd(fIndet),dias:dPara});
        }
        // contrato por vencer (fecha termino dentro de X dias)
        var fin = _aFecha(r[C.COL_TERMINO]);
        if(fin){
          var dVence = Math.round((fin.getTime()-hoy.getTime())/86400000);
          if(dVence>=0 && dVence<=C.POR_VENCER_DIAS) porVencer.push({dni:dni,nombre:nombre,empresa:emp,fecha:_ymd(fin),dias:dVence});
        }
      });
    });
    proximosIndet.sort(function(a,b){return a.dias-b.dias;});
    porVencer.sort(function(a,b){return a.dias-b.dias;});
    return { ok:true, total:total, porTipo:porTipo, porEmpresa:porEmpresa, proximosIndet:proximosIndet, porVencer:porVencer, generado:new Date().toISOString() };
  }catch(err){ return { ok:false, message:String(err) }; }
}

    function bucket(m){ if(m==null) return 'sinDato'; if(m<3) return 'm3'; if(m<12) return 'm3a12'; if(m<36) return 'a1a3'; if(m<54) return 'a3a54'; return 'mas54'; }
    function evaluar(modulo, dni){
      dni=String(dni||'').trim(); if(!dni) return;
      var p = padron[dni] || {tipo:'sinDato', meses:null};
      porTipo[p.tipo]=(porTipo[p.tipo]||0)+1;
      porAntig[bucket(p.meses)]++;
      if(!porModulo[modulo]) porModulo[modulo]={prueba:0,plazoFijo:0,indeterminado:0,sinDato:0};
      porModulo[modulo][p.tipo]++;
    }

    if(ver.accidentes){ var a=leer('BS_Accidentes_Casos'); var aDni=ix(a.h,'dni'); a.rows.forEach(function(r){ evaluar('Accidentes', r[aDni]); }); }
    if(ver.hostigamiento){ var h=leer('BS_Hostigamiento'); var hDte=ix(h.h,'denunciante_dni'), hDdo=ix(h.h,'denunciado_dni');
      h.rows.forEach(function(r){ evaluar('Hostigamiento', r[hDte]); evaluar('Hostigamiento', r[hDdo]); }); }
    if(ver.subsidios){ var s=leer('BS_Subsidios'); var sNum=ix(s.h,'numero'); s.rows.forEach(function(r){ evaluar('Subsidios', r[sNum]); }); }

    var total = porTipo.prueba+porTipo.plazoFijo+porTipo.indeterminado+porTipo.sinDato;
    return { ok:true, total:total, porTipo:porTipo, porAntiguedad:porAntig, porModulo:porModulo, generado:new Date().toISOString() };
  }catch(err){ return { ok:false, message:String(err) }; }
}

/**
 * ============================================================
 *  bs_listarBuzones  (Q2a)
 * ============================================================
 *  Pegar al FINAL del codigo GAS de Bienestar Social.
 *  ADITIVO: nueva funcion + un (1) case en el router.
 *  Lee la hoja BS_Buzones (creada por bs_sembrarBuzones) y
 *  devuelve los buzones agrupados por sector.
 *
 *  Mapea por NOMBRE de encabezado (no por indice), asi no
 *  importa el orden de columnas.
 *
 *  Respuesta (OBJETO):
 *  { ok:true, total:46, sectores:[
 *      { empresa:'VERFRUT', sector:'PUNTA ARENA', cantidad:8,
 *        ubicaciones:[ {numero:1, ubicacion:'GARITA ORGANICOS'}, ... ] }, ...
 *  ]}
 *
 *  Param opcional:  params.empresa = 'RAPEL' | 'VERFRUT'  (vacio = todas)
 * ============================================================
 */
function bs_listarBuzones(params) {
  try {
    params = params || {};
    var fEmpresa = (params.empresa || '').toString().trim().toUpperCase();
 
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sh = ss.getSheetByName('BS_Buzones');
    if (!sh) {
      return { ok: false, message: 'No existe la hoja BS_Buzones. Ejecuta bs_sembrarBuzones primero.' };
    }
 
    var valores = sh.getDataRange().getValues();
    if (valores.length < 2) return { ok: true, total: 0, sectores: [] };
 
    // indice de columnas por encabezado
    var headers = valores[0];
    var col = {};
    for (var c = 0; c < headers.length; c++) {
      col[String(headers[c]).trim().toLowerCase()] = c;
    }
    var iEmp = col['empresa'], iSec = col['sector'],
        iNum = col['numero'], iUbi = col['ubicacion'], iAct = col['activo'];
 
    var orden = [];   // preserva orden de aparicion de los sectores
    var mapa = {};
    var total = 0;
 
    for (var r = 1; r < valores.length; r++) {
      var fila = valores[r];
      var emp = String(fila[iEmp] || '').trim();
      var sec = String(fila[iSec] || '').trim();
      if (!emp && !sec) continue;
      if (iAct !== undefined && fila[iAct] === false) continue; // buzon desactivado
      if (fEmpresa && emp.toUpperCase() !== fEmpresa) continue;
 
      var clave = emp + '|' + sec;
      if (!mapa[clave]) {
        mapa[clave] = { empresa: emp, sector: sec, cantidad: 0, ubicaciones: [] };
        orden.push(clave);
      }
      mapa[clave].ubicaciones.push({
        numero: (iNum !== undefined ? fila[iNum] : mapa[clave].cantidad + 1),
        ubicacion: String(fila[iUbi] || '').trim()
      });
      mapa[clave].cantidad++;
      total++;
    }
 
    var sectores = [];
    for (var k = 0; k < orden.length; k++) sectores.push(mapa[orden[k]]);
 
    return { ok: true, total: total, sectores: sectores };
  } catch (err) {
    return { ok: false, message: 'bs_listarBuzones: ' + err.message };
  }
}

// Listar (filtros opcionales: estado, usuario, modulo). Ordena por id desc.
function bs_listarSugerencias(filtros) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName('BS_Sugerencias');
    if (!sheet) return { ok: true, items: [], total: 0 };
    var data = sheet.getDataRange().getValues();
    if (data.length <= 1) return { ok: true, items: [], total: 0 };
    var headers = data[0];
    var items = data.slice(1).map(function(row) {
      var o = {};
      headers.forEach(function(h, i) { var val = row[i]; if (val instanceof Date) val = val.toISOString(); o[h] = val; });
      return o;
    });
    filtros = filtros || {};
    if (filtros.estado)  items = items.filter(function(i){ return i.estado === filtros.estado; });
    if (filtros.usuario) items = items.filter(function(i){ return i.usuario === filtros.usuario; });
    if (filtros.modulo)  items = items.filter(function(i){ return i.modulo === filtros.modulo; });
    items.sort(function(a, b) { return (b.id || 0) - (a.id || 0); });
    return { ok: true, items: items, total: items.length };
  } catch (err) { return { ok: false, message: err.message }; }
}
 
// Actualizar (solo admin desde el front). Sella fecha_finalizado al pasar a Finalizada.
function bs_actualizarSugerencia(usuario, id, datos) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName('BS_Sugerencias');
    if (!sheet) return { ok: false, message: 'Hoja BS_Sugerencias no existe' };
    var data = sheet.getDataRange().getValues();
    var headers = data[0];
    var updatables = ['titulo','modulo','descripcion','prioridad','estado','respuesta_admin'];
    datos = datos || {};
    for (var i = 1; i < data.length; i++) {
      if (parseInt(data[i][0]) === parseInt(id)) {
        var rowNum = i + 1;
        updatables.forEach(function(field) {
          if (datos[field] !== undefined) {
            var col = headers.indexOf(field);
            if (col >= 0) sheet.getRange(rowNum, col + 1).setValue(datos[field]);
          }
        });
        if (datos.estado === 'Finalizada') {
          var colF = headers.indexOf('fecha_finalizado');
          if (colF >= 0) sheet.getRange(rowNum, colF + 1).setValue(new Date());
        }
        var colU = headers.indexOf('updated_at');
        if (colU >= 0) sheet.getRange(rowNum, colU + 1).setValue(new Date());
        return { ok: true, message: 'Sugerencia actualizada' };
      }
    }
    return { ok: false, message: 'Sugerencia no encontrada' };
  } catch (err) { return { ok: false, message: err.message }; }
}

// Parser robusto de fecha 'YYYY-MM-DD' (o Date) a fecha LOCAL sin desfase de zona horaria.
function _segParseFecha(v) {
  if (!v) return null;
  var str;
  if (Object.prototype.toString.call(v) === '[object Date]') {
    str = Utilities.formatDate(v, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  } else {
    str = String(v).substring(0, 10);
  }
  var p = str.split('-');
  if (p.length !== 3) return null;
  var d = new Date(parseInt(p[0], 10), parseInt(p[1], 10) - 1, parseInt(p[2], 10));
  d.setHours(0, 0, 0, 0);
  return isNaN(d.getTime()) ? null : d;
}
 
// Revisa BS_Quejas y envia un correo a ADMIN_EMAILS con los seguimientos vencidos / por vencer.
function bs_alertasSeguimientoQuejas() {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sh = ss.getSheetByName('BS_Quejas');
    if (!sh) return;
    var data = sh.getDataRange().getValues();
    if (data.length <= 1) return;
    var H = data[0].map(function(h){ return String(h).trim(); });
    var ci = {};
    H.forEach(function(h, i){ ci[h] = i; });
    if (ci['seguimiento'] === undefined || ci['seguimiento_fin'] === undefined) return;
 
    var hoy = new Date(); hoy.setHours(0, 0, 0, 0);
    var vencidos = [], porVencer = [];
 
    for (var r = 1; r < data.length; r++) {
      var row = data[r];
      if (String(row[ci['seguimiento']]).trim() !== 'Sí') continue;
      if (ci['resultado'] !== undefined && String(row[ci['resultado']]).trim() === 'Cerrado') continue;
      var fin = _segParseFecha(row[ci['seguimiento_fin']]);
      if (!fin) continue;
      var dias = Math.round((fin - hoy) / 86400000);
      var alerta = (ci['seguimiento_alerta'] !== undefined ? parseInt(row[ci['seguimiento_alerta']], 10) : 1) || 1;
      var it = {
        id: row[ci['id']],
        empresa: ci['empresa'] !== undefined ? row[ci['empresa']] : '',
        sector: ci['sector'] !== undefined ? row[ci['sector']] : '',
        frec: ci['seguimiento_frecuencia'] !== undefined ? row[ci['seguimiento_frecuencia']] : '',
        fin: Utilities.formatDate(fin, Session.getScriptTimeZone(), 'dd/MM/yyyy'),
        dias: dias
      };
      if (dias < 0) vencidos.push(it);
      else if (dias <= alerta) porVencer.push(it);
    }
 
    if (!vencidos.length && !porVencer.length) return;
 
    var to = (ALERTAS_CONFIG.ADMIN_EMAILS || []).join(',');
    if (!to) return;
 
    function filas(arr, color) {
      return arr.map(function(it){
        var d = it.dias < 0 ? ('Vencido hace ' + Math.abs(it.dias) + ' dia(s)')
              : (it.dias === 0 ? 'Vence hoy' : ('Vence en ' + it.dias + ' dia(s)'));
        return '<tr>' +
          '<td style="padding:4px 8px;border:1px solid #e5e7eb;">#' + it.id + '</td>' +
          '<td style="padding:4px 8px;border:1px solid #e5e7eb;">' + (it.empresa || '') + ' / ' + (it.sector || '') + '</td>' +
          '<td style="padding:4px 8px;border:1px solid #e5e7eb;">' + (it.frec || '') + '</td>' +
          '<td style="padding:4px 8px;border:1px solid #e5e7eb;">' + it.fin + '</td>' +
          '<td style="padding:4px 8px;border:1px solid #e5e7eb;color:' + color + ';font-weight:600;">' + d + '</td>' +
        '</tr>';
      }).join('');
    }
    var th = '<tr><th style="padding:4px 8px;border:1px solid #e5e7eb;">ID</th>' +
             '<th style="padding:4px 8px;border:1px solid #e5e7eb;">Empresa / Sector</th>' +
             '<th style="padding:4px 8px;border:1px solid #e5e7eb;">Frecuencia</th>' +
             '<th style="padding:4px 8px;border:1px solid #e5e7eb;">Vence</th>' +
             '<th style="padding:4px 8px;border:1px solid #e5e7eb;">Estado</th></tr>';
 
    var html = '<div style="font-family:Segoe UI,Arial,sans-serif;color:#0b1e45">' +
      '<h2 style="color:#0033A0;margin:0 0 10px">Seguimiento de quejas - alertas</h2>';
    if (vencidos.length) {
      html += '<h3 style="color:#dc2626;margin:12px 0 4px">Vencidos (' + vencidos.length + ')</h3>' +
        '<table style="border-collapse:collapse;font-size:13px">' + th + filas(vencidos, '#dc2626') + '</table>';
    }
    if (porVencer.length) {
      html += '<h3 style="color:#d97706;margin:12px 0 4px">Por vencer (' + porVencer.length + ')</h3>' +
        '<table style="border-collapse:collapse;font-size:13px">' + th + filas(porVencer, '#d97706') + '</table>';
    }
    html += '<p style="margin:14px 0 0"><a href="https://joeltimoteog-bot.github.io/bienestar-social-verfrut/modulos/quejas.html" style="background:#0033A0;color:#fff;padding:10px 16px;border-radius:8px;text-decoration:none">Ver quejas</a></p></div>';
 
    MailApp.sendEmail({
      to: to,
      subject: 'Seguimiento de quejas: ' + vencidos.length + ' vencidos, ' + porVencer.length + ' por vencer',
      htmlBody: html
    });
  } catch (e) {
    Logger.log('bs_alertasSeguimientoQuejas error: ' + e.message);
  }
}

// ---- Parser de fecha tolerante (Date o 'YYYY-MM-DD'), a fecha LOCAL sin desfase ----
function _estParseFecha(v) {
  if (v === null || v === undefined || v === '') return null;
  if (Object.prototype.toString.call(v) === '[object Date]') {
    if (isNaN(v.getTime())) return null;
    var d = new Date(v.getFullYear(), v.getMonth(), v.getDate()); d.setHours(0,0,0,0); return d;
  }
  var s = String(v).trim(); if (!s) return null;
  var m = s.match(/^(\d{4})-(\d{2})-(\d{2})/);
  if (m) { var d2 = new Date(parseInt(m[1],10), parseInt(m[2],10)-1, parseInt(m[3],10)); d2.setHours(0,0,0,0); return d2; }
  var d3 = new Date(s); if (isNaN(d3.getTime())) return null;
  var d4 = new Date(d3.getFullYear(), d3.getMonth(), d3.getDate()); d4.setHours(0,0,0,0); return d4;
}
 
function _estLeerParams() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName(BS_EST_HOJA_PARAMS);
  if (!sh || sh.getLastRow() < 2) return [];
  var data = sh.getDataRange().getValues();
  var H = data[0].map(function(x){ return String(x).trim(); });
  var idx = {}; H.forEach(function(h,i){ idx[h] = i; });
  var out = [];
  for (var r = 1; r < data.length; r++) {
    var row = data[r];
    var modulo = String(row[idx['modulo']] || '').trim();
    if (!modulo) continue;
    out.push({
      modulo: modulo,
      hoja: String(row[idx['hoja']] || '').trim(),
      col_estado: String(row[idx['col_estado']] || '').trim(),
      estados_cierre: String(row[idx['estados_cierre']] || '').split(',').map(function(s){ return s.trim().toLowerCase(); }).filter(String),
      col_fecha_inicio: String(row[idx['col_fecha_inicio']] || '').trim(),
      col_fecha_cierre: String(row[idx['col_fecha_cierre']] || '').trim(),
      col_responsable: (String(row[idx['col_responsable']] || '').trim() || 'created_by'),
      plazo_dias: parseInt(row[idx['plazo_dias']], 10) || 0,
      usa_seguimiento: String(row[idx['usa_seguimiento']] || '').trim().toUpperCase() === 'SI'
    });
  }
  return out;
}
 
// ---- Endpoint principal ----
function bs_estadisticasEmpresariales(params) {
  try {
    params = params || {};
    var fModulo  = (params.modulo  && params.modulo  !== 'Todos')  ? params.modulo  : null;
    var fUsuario = (params.usuario && params.usuario !== 'Todos')  ? params.usuario : null;
    var fEmpresa = (params.empresa && params.empresa !== 'Todas')  ? params.empresa : null;
    var fDesde   = params.desde ? _estParseFecha(params.desde) : null;
    var fHasta   = params.hasta ? _estParseFecha(params.hasta) : null;
    var hoy = new Date(); hoy.setHours(0,0,0,0);
 
    var cfgs = _estLeerParams();
    if (!cfgs.length) return { ok:false, message:'BS_Parametros vacia. Ejecuta bs_setupParametros.' };
 
    var ss = SpreadsheetApp.getActiveSpreadsheet();
 
    // Cargar cada hoja una sola vez
    var cache = {};
    cfgs.forEach(function(cfg){
      var sh = ss.getSheetByName(cfg.hoja);
      if (!sh || sh.getLastRow() < 2 || sh.getLastColumn() < 1) { cache[cfg.modulo] = { ci:{}, rows:[] }; return; }
      var data = sh.getDataRange().getValues();
      var H = data[0].map(function(x){ return String(x).trim(); });
      var ci = {}; H.forEach(function(h,i){ ci[h] = i; });
      cache[cfg.modulo] = { ci:ci, rows:data.slice(1) };
    });
 
    // Usuarios disponibles (sin filtros) para el desplegable
    var usuariosSet = {};
    cfgs.forEach(function(cfg){
      var c = cache[cfg.modulo]; var cb = c.ci['created_by'];
      if (cb === undefined) return;
      c.rows.forEach(function(row){ var u = String(row[cb]||'').trim(); if (u) usuariosSet[u] = true; });
    });
 
    var totales = {}, estados = {}, porEmpresa = { RAPEL:{}, VERFRUT:{} }, tendMap = {};
    var usuariosMap = {}, cumpModulo = {}, cumpUsuario = {}, vencidos = [];
    var generalTotal = 0;
    function nuevoCump(){ return { total:0, cerrados:0, a_tiempo:0, fuera:0, abierto_plazo:0, vencido:0, suma_dias_cierre:0, n_cierre:0 }; }
    var cumpGlobal = nuevoCump();
 
    cfgs.forEach(function(cfg){
      if (fModulo && cfg.modulo !== fModulo) return;
      var c = cache[cfg.modulo], ci = c.ci;
      totales[cfg.modulo] = totales[cfg.modulo] || 0;
      estados[cfg.modulo] = estados[cfg.modulo] || {};
      cumpModulo[cfg.modulo] = cumpModulo[cfg.modulo] || nuevoCump();
 
      var cEstado = ci[cfg.col_estado];
      var cIni    = ci[cfg.col_fecha_inicio];
      var cCierre = cfg.col_fecha_cierre ? ci[cfg.col_fecha_cierre] : undefined;
      var cResp   = ci[cfg.col_responsable] !== undefined ? ci[cfg.col_responsable] : ci['created_by'];
      var cEmp    = ci['empresa'];
      var cCreated= ci['created_at'];
      var cUpd    = ci['updated_at'];
      var cSeg    = ci['seguimiento'];
      var cSegFin = ci['seguimiento_fin'];
      var cCb     = ci['created_by'];
      var cId     = ci['id'];
 
      c.rows.forEach(function(row){
        var cbVal = cCb !== undefined ? String(row[cCb]||'').trim() : '';
        var respVal = cResp !== undefined ? String(row[cResp]||'').trim() : cbVal;
        var emp = cEmp !== undefined ? String(row[cEmp]||'').trim() : '';
        var fIni = cIni !== undefined ? _estParseFecha(row[cIni]) : null;
        if (!fIni && cCreated !== undefined) fIni = _estParseFecha(row[cCreated]);
 
        // Filtros
        if (fUsuario && respVal !== fUsuario && cbVal !== fUsuario) return;
        if (fEmpresa && emp !== fEmpresa) return;
        if (fDesde && fIni && fIni < fDesde) return;
        if (fHasta && fIni && fIni > fHasta) return;
 
        // Conteos
        totales[cfg.modulo]++; generalTotal++;
        var est = (cEstado !== undefined ? String(row[cEstado]||'').trim() : '') || '(sin estado)';
        estados[cfg.modulo][est] = (estados[cfg.modulo][est]||0) + 1;
        if (emp === 'RAPEL' || emp === 'VERFRUT') porEmpresa[emp][cfg.modulo] = (porEmpresa[emp][cfg.modulo]||0) + 1;
        if (fIni) {
          var ym = fIni.getFullYear() + '-' + ('0'+(fIni.getMonth()+1)).slice(-2);
          tendMap[ym] = tendMap[ym] || {}; tendMap[ym][cfg.modulo] = (tendMap[ym][cfg.modulo]||0) + 1;
        }
        var ukey = cbVal || respVal || '(sin usuario)';
        usuariosMap[ukey] = usuariosMap[ukey] || { total:0, porModulo:{} };
        usuariosMap[ukey].total++; usuariosMap[ukey].porModulo[cfg.modulo] = (usuariosMap[ukey].porModulo[cfg.modulo]||0) + 1;
 
        // Cumplimiento
        var esCierre = cfg.estados_cierre.length > 0 && cfg.estados_cierre.indexOf(est.toLowerCase()) >= 0;
        var limite = null;
        if (cfg.usa_seguimiento && cSeg !== undefined && String(row[cSeg]||'').trim() === 'Sí' && cSegFin !== undefined) {
          limite = _estParseFecha(row[cSegFin]);
        }
        if (!limite && fIni && cfg.plazo_dias > 0) { limite = new Date(fIni.getTime()); limite.setDate(limite.getDate() + cfg.plazo_dias); }
 
        var CM = cumpModulo[cfg.modulo];
        cumpUsuario[ukey] = cumpUsuario[ukey] || nuevoCump();
        var CU = cumpUsuario[ukey];
        CM.total++; CU.total++; cumpGlobal.total++;
 
        if (esCierre) {
          CM.cerrados++; CU.cerrados++; cumpGlobal.cerrados++;
          var fCierre = (cCierre !== undefined ? _estParseFecha(row[cCierre]) : null) || (cUpd !== undefined ? _estParseFecha(row[cUpd]) : null);
          var aTiempo = (limite && fCierre) ? (fCierre <= limite) : true;
          if (aTiempo) { CM.a_tiempo++; CU.a_tiempo++; cumpGlobal.a_tiempo++; } else { CM.fuera++; CU.fuera++; cumpGlobal.fuera++; }
          if (fIni && fCierre) { var dd = Math.round((fCierre - fIni)/86400000); if (dd >= 0) { CM.suma_dias_cierre += dd; CM.n_cierre++; CU.suma_dias_cierre += dd; CU.n_cierre++; cumpGlobal.suma_dias_cierre += dd; cumpGlobal.n_cierre++; } }
        } else if (limite && hoy > limite) {
          CM.vencido++; CU.vencido++; cumpGlobal.vencido++;
          vencidos.push({ modulo:cfg.modulo, id:(cId!==undefined?row[cId]:''), usuario:ukey, empresa:emp,
            fecha_inicio: fIni ? Utilities.formatDate(fIni, Session.getScriptTimeZone(), 'dd/MM/yyyy') : '',
            limite: Utilities.formatDate(limite, Session.getScriptTimeZone(), 'dd/MM/yyyy'),
            dias_vencido: Math.round((hoy - limite)/86400000) });
        } else {
          CM.abierto_plazo++; CU.abierto_plazo++; cumpGlobal.abierto_plazo++;
        }
      });
    });
 
    function cerrarCump(c){
      var cumplidos = c.a_tiempo + c.abierto_plazo;
      c.pct_cierre = c.total ? Math.round(c.cerrados*1000/c.total)/10 : 0;
      c.pct_cumplimiento = c.total ? Math.round(cumplidos*1000/c.total)/10 : 0;
      c.pct_a_tiempo = c.cerrados ? Math.round(c.a_tiempo*1000/c.cerrados)/10 : 0;
      c.tiempo_prom_cierre = c.n_cierre ? Math.round(c.suma_dias_cierre*10/c.n_cierre)/10 : 0;
      delete c.suma_dias_cierre; delete c.n_cierre;
      return c;
    }
    cerrarCump(cumpGlobal);
    Object.keys(cumpModulo).forEach(function(k){ cerrarCump(cumpModulo[k]); });
    var porUsuario = Object.keys(cumpUsuario).map(function(k){ var c = cerrarCump(cumpUsuario[k]); c.usuario = k; return c; });
    porUsuario.sort(function(a,b){ return b.pct_cumplimiento - a.pct_cumplimiento; });
    vencidos.sort(function(a,b){ return b.dias_vencido - a.dias_vencido; });
 
    var ymKeys = Object.keys(tendMap).sort();
    var modList = cfgs.filter(function(cf){ return !fModulo || cf.modulo === fModulo; }).map(function(cf){ return cf.modulo; });
    var tendencia = { labels: ymKeys, series: {} };
    modList.forEach(function(m){ tendencia.series[m] = ymKeys.map(function(k){ return tendMap[k][m] || 0; }); });
 
    var usuarios = Object.keys(usuariosMap).map(function(k){ return { usuario:k, total:usuariosMap[k].total, porModulo:usuariosMap[k].porModulo }; });
    usuarios.sort(function(a,b){ return b.total - a.total; });
 
    return {
      ok: true,
      generado: Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm'),
      filtros: { modulo: fModulo||'Todos', usuario: fUsuario||'Todos', empresa: fEmpresa||'Todas', desde: params.desde||'', hasta: params.hasta||'' },
      totales: totales,
      general: generalTotal,
      porEmpresa: porEmpresa,
      estados: estados,
      tendencia: tendencia,
      usuarios: usuarios,
      usuariosDisponibles: Object.keys(usuariosSet).sort(),
      cumplimiento: { global: cumpGlobal, porModulo: cumpModulo, porUsuario: porUsuario, vencidosAbiertos: vencidos.slice(0, 100) }
    };
  } catch (e) {
    return { ok:false, message:'Error: ' + e.message };
  }
}

function _bsWhatsApp(telefono, apikey, mensaje) {
  try {
    if (!apikey) { Logger.log('Sin apikey para ' + telefono + ' (se omite)'); return false; }
    var url = 'https://api.callmebot.com/whatsapp.php'
            + '?phone='  + encodeURIComponent(telefono)
            + '&text='   + encodeURIComponent(mensaje)
            + '&apikey=' + encodeURIComponent(apikey);
    var resp = UrlFetchApp.fetch(url, { muteHttpExceptions: true });
    Logger.log('WhatsApp ' + telefono + ' -> ' + resp.getResponseCode());
    return true;
  } catch (e) {
    Logger.log('WhatsApp error (' + telefono + '): ' + e.message);
    return false;
  }
}

const BS_HOJA_HISTORIAL_ACC = 'BS_Accidentes_Historial';
const BS_HIST_ACC_COLS = [
  'timestamp', 'id_caso', 'dni', 'nombre', 'tipo_cambio', 'usuario',
  'descanso_medico', 'dm_inicio', 'dm_fin', 'dm_dias',
  'formulario_con_goce', 'goce_inicio', 'goce_fin', 'goce_dias',
  'estado_caso', 'detalle_json'
];
 
function _bsHistSheet_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName(BS_HOJA_HISTORIAL_ACC);
  if (!sh) {
    sh = ss.insertSheet(BS_HOJA_HISTORIAL_ACC);
    sh.getRange(1, 1, 1, BS_HIST_ACC_COLS.length).setValues([BS_HIST_ACC_COLS]);
    sh.setFrozenRows(1);
  }
  return sh;
}

/**
 * Registra un movimiento en el historial. NUNCA lanza error (no rompe el guardado).
 * @param {number|string} idCaso  id del caso
 * @param {Object} datos          objeto con los campos del caso/formulario
 * @param {string} usuario        quién hace el cambio
 * @param {string} tipoCambio     'CREACION' | 'ACTUALIZACION' | 'DM' | 'FORMULARIO'
 */
function bs_logHistorialAccidente(idCaso, datos, usuario, tipoCambio) {
  try {
    const sh = _bsHistSheet_();
    const d = datos || {};
    sh.appendRow([
      new Date(),
      idCaso || d.id || '',
      d.dni || '',
      d.nombre || '',
      tipoCambio || 'ACTUALIZACION',
      usuario || d.created_by || '',
      d.descanso_medico || '',
      d.descanso_medico_inicio || '',
      d.descanso_medico_fin || '',
      d.descanso_medico_dias || '',
      d.formulario_con_goce || '',
      d.goce_inicio || '',
      d.goce_fin || '',
      d.goce_dias || '',
      d.estado_caso || '',
      JSON.stringify(d)
    ]);
  } catch (err) {
    Logger.log('bs_logHistorialAccidente ERROR: ' + err);
  }
}
 
/**

  try {

    const sh = ss.getSheetByName(BS_HOJA_HISTORIAL_ACC);
    if (!sh || sh.getLastRow() < 2) return { ok: true, items: [] };
    const vals = sh.getRange(1, 1, sh.getLastRow(), sh.getLastColumn()).getValues();
    const headers = vals[0];
    const items = [];
    for (let i = 1; i < vals.length; i++) {
      const o = {};
      for (let j = 0; j < headers.length; j++) { o[headers[j]] = vals[i][j]; }
      items.push(o);

    if (params && params.id_caso) {
      const idc = String(params.id_caso);
      return { ok: true, items: items.filter(function (x) { return String(x.id_caso) === idc; }) };
    }
    return { ok: true, items: items };
  } catch (err) {
    return { ok: false, error: String(err), items: [] };
  }
}

function _bsHojaSeguimientos_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  // 1) Constante de config, si existe (p. ej. BS_CONFIG.HOJA_ACC_SEGUIMIENTOS)
  try {
    if (typeof BS_CONFIG !== 'undefined' && BS_CONFIG && BS_CONFIG.HOJA_ACC_SEGUIMIENTOS) {
      const s1 = ss.getSheetByName(BS_CONFIG.HOJA_ACC_SEGUIMIENTOS);
      if (s1) return s1;
    }
  } catch (e) {}
  // 2) Nombre literal habitual
  const s2 = ss.getSheetByName('BS_Accidentes_Seguimientos');
  if (s2) return s2;
  // 3) Cualquier hoja cuyo nombre relacione accidentes + seguimientos
  const all = ss.getSheets();
  for (let i = 0; i < all.length; i++) {
    const n = all[i].getName();
    if (/seguimiento/i.test(n) && /accidente/i.test(n)) return all[i];
  }
  return null;
}

    if (sh.getLastRow() < 2) return { ok: true, items: [] };
    const vals = sh.getRange(1, 1, sh.getLastRow(), sh.getLastColumn()).getValues();
    const headers = vals[0];
    const items = [];
    for (let i = 1; i < vals.length; i++) {
      const o = {};
      for (let j = 0; j < headers.length; j++) { o[headers[j]] = vals[i][j]; }
      items.push(o);
    }
    // opcional: filtrar por un caso
    if (params && params.id_caso) {
      const idc = String(params.id_caso);
      return { ok: true, items: items.filter(function (x) { return String(x.id_caso) === idc; }) };
    }
    return { ok: true, items: items };
  } catch (err) {
    return { ok: false, error: String(err), items: [] };
  }
}

/** Lista los seguimientos de un hallazgo (id por POSICIÓN, igual que Accidentes). */
function bs_listarSeguimientosQueja(id_queja) {
  try {
    var sheet = _bsQSegSheet_();
    var lr = sheet.getLastRow();
    if (lr < 2) return { ok: true, items: [] };
    var lc = sheet.getLastColumn();
    var headers = sheet.getRange(1, 1, 1, lc).getValues()[0];
    var data = sheet.getRange(2, 1, lr - 1, lc).getValues();
    var iIdQ = headers.indexOf('id_queja');
    var target = String(id_queja).trim();
    var items = [];
    for (var r = 0; r < data.length; r++) {
      if (String(data[r][iIdQ]).trim() !== target) continue;
      var o = {};
      for (var c = 0; c < headers.length; c++) o[headers[c]] = data[r][c];
      items.push(o);
    }
    items.sort(function(a, b) { return (Number(a.numero_seguimiento) || 0) - (Number(b.numero_seguimiento) || 0); });
    return { ok: true, items: items };
  } catch (err) { return { ok: false, message: err.message }; }
}
 
/** Agrega un seguimiento.  datos = { fecha_seguimiento, detalle } */
function bs_crearSeguimientoQueja(usuario, id_queja, datos) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = _bsQSegSheet_();
    datos = datos || {};
    if (!datos.fecha_seguimiento || !datos.detalle) return { ok: false, message: 'Completa fecha y detalle' };
 
    const lastRow = sheet.getLastRow();
    const id = lastRow > 1 ? (parseInt(sheet.getRange(lastRow, 1).getValue()) || 0) + 1 : 1;
 
    // numero_seguimiento (consecutivo dentro del hallazgo)
    const existentes = bs_listarSeguimientosQueja(id_queja);
    const numero = (existentes.items ? existentes.items.length : 0) + 1;
 
    sheet.appendRow([
      id,
      parseInt(id_queja),
      numero,
      datos.fecha_seguimiento ? new Date(datos.fecha_seguimiento) : new Date(),
      datos.detalle || '',
      usuario,
      new Date()
    ]);
 
    // Refrescar updated_at del hallazgo padre (solo si la columna existe; NO toca cantidad_quejas)
    const sheetQ = ss.getSheetByName('BS_Quejas');
    if (sheetQ) {
      const dataQ = sheetQ.getDataRange().getValues();
      const hdrQ = dataQ[0];
      for (let i = 1; i < dataQ.length; i++) {
        if (parseInt(dataQ[i][0]) === parseInt(id_queja)) {
          const colUpd = hdrQ.indexOf('updated_at');
          if (colUpd >= 0) sheetQ.getRange(i + 1, colUpd + 1).setValue(new Date());
          break;
        }
      }
    }
 
    return { ok: true, id: id, numero: numero, message: 'Seguimiento #' + numero + ' registrado' };
  } catch (err) { return { ok: false, message: err.message }; }
}

// Crear (header-based: mapea por nombre de encabezado, no por posición).
function bs_crearRendicion(usuario, datos) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName(BS_REND_SHEET);
    if (!sheet) return { ok: false, message: 'Hoja BS_Rendiciones no existe. Ejecuta bs_setupRendiciones.' };
    var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0].map(function(h){ return String(h).trim(); });
    var lastRow = sheet.getLastRow();
    var id = lastRow > 1 ? (parseInt(sheet.getRange(lastRow, 1).getValue()) || 0) + 1 : 1;
    var now = new Date();
    datos = datos || {};
    var fecha = datos.fecha ? new Date(datos.fecha) : now;
    var v = {
      id: id,
      fecha: fecha,
      semana: parseInt(datos.semana) || '',
      anio: parseInt(datos.anio) || (fecha ? fecha.getFullYear() : now.getFullYear()),
      rendicion: datos.rendicion || '',
      empresa: datos.empresa || '',
      monto_reclutamiento: parseFloat(datos.monto_reclutamiento) || 0,
      monto_relaciones_laborales: parseFloat(datos.monto_relaciones_laborales) || 0,
      monto_capacitaciones: parseFloat(datos.monto_capacitaciones) || 0,
      monto_bienestar: parseFloat(datos.monto_bienestar) || 0,
      monto_total: parseFloat(datos.monto_total) || 0,
      responsable: datos.responsable || '',
      estado: datos.estado || 'NO DEPOSITADO',
      fecha_deposito: datos.fecha_deposito ? new Date(datos.fecha_deposito) : '',
      observaciones: datos.observaciones || '',
      created_by: usuario || '',
      created_at: now,
      updated_at: now
    };
    var fila = headers.map(function(h){ return (v[h] !== undefined) ? v[h] : ''; });
    sheet.appendRow(fila);
    return { ok: true, id: id, message: 'Rendición registrada' };
  } catch (err) { return { ok: false, message: err.message }; }
}
 
// Listar (filtros opcionales: anio, empresa, responsable, estado). Ordena por id desc.
function bs_listarRendiciones(filtros) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName(BS_REND_SHEET);
    if (!sheet) return { ok: false, message: 'Hoja BS_Rendiciones no existe' };
    var data = sheet.getDataRange().getValues();
    if (data.length <= 1) return { ok: true, items: [], total: 0 };
    var headers = data[0];
    var items = data.slice(1).map(function(row) {
      var obj = {};
      headers.forEach(function(h, i) { var v = row[i]; if (v instanceof Date) v = v.toISOString(); obj[String(h).trim()] = v; });
      return obj;
    });
    filtros = filtros || {};
    if (filtros.anio)        items = items.filter(function(i){ return String(i.anio) === String(filtros.anio); });
    if (filtros.empresa)     items = items.filter(function(i){ return i.empresa === filtros.empresa; });
    if (filtros.responsable) items = items.filter(function(i){ return i.responsable === filtros.responsable; });
    if (filtros.estado)      items = items.filter(function(i){ return String(i.estado).toUpperCase() === String(filtros.estado).toUpperCase(); });
    items.sort(function(a, b) { return (b.id || 0) - (a.id || 0); });
    return { ok: true, items: items, total: items.length };
  } catch (err) { return { ok: false, message: err.message }; }
}

// Reporte consolidado (KPIs + por empresa + por categoría + por responsable + tendencia mensual).
function bs_reporteRendiciones(filtros) {
  try {
    var r = bs_listarRendiciones(filtros);
    if (!r.ok) return r;
    var items = r.items;
    var cat = { reclutamiento:0, relaciones_laborales:0, capacitaciones:0, bienestar:0 };
    var porEmpresa = {
      RAPEL:   { total:0, reclutamiento:0, relaciones_laborales:0, capacitaciones:0, bienestar:0, n:0 },
      VERFRUT: { total:0, reclutamiento:0, relaciones_laborales:0, capacitaciones:0, bienestar:0, n:0 }
    };
    var porResponsable = {}, porMes = {};
    var total = 0, depMonto = 0, penMonto = 0, nDep = 0, nPen = 0;
    function emp(v){ var s = String(v||'').toUpperCase(); return s.indexOf('VERFRUT')>=0 ? 'VERFRUT' : (s.indexOf('RAPEL')>=0 ? 'RAPEL' : null); }
 
    items.forEach(function(it){
      var mt  = parseFloat(it.monto_total) || 0;
      var rec = parseFloat(it.monto_reclutamiento) || 0;
      var rll = parseFloat(it.monto_relaciones_laborales) || 0;
      var cap = parseFloat(it.monto_capacitaciones) || 0;
      var bie = parseFloat(it.monto_bienestar) || 0;
      total += mt; cat.reclutamiento += rec; cat.relaciones_laborales += rll; cat.capacitaciones += cap; cat.bienestar += bie;
 
      if (String(it.estado).toUpperCase() === 'DEPOSITADO') { depMonto += mt; nDep++; }
      else { penMonto += mt; nPen++; }
 
      var e = emp(it.empresa);
      if (e) { var o = porEmpresa[e]; o.total+=mt; o.reclutamiento+=rec; o.relaciones_laborales+=rll; o.capacitaciones+=cap; o.bienestar+=bie; o.n++; }
 
      var rs = String(it.responsable || '(sin responsable)').trim();
      if (!porResponsable[rs]) porResponsable[rs] = { total:0, n:0 };
      porResponsable[rs].total += mt; porResponsable[rs].n++;
 
      var ym = '';
      if (it.fecha) { var d = new Date(it.fecha); if (!isNaN(d.getTime())) ym = d.getFullYear() + '-' + ('0'+(d.getMonth()+1)).slice(-2); }
      if (ym) { porMes[ym] = (porMes[ym] || 0) + mt; }
    });
 
    var responsables = Object.keys(porResponsable).map(function(k){ return { responsable:k, total:porResponsable[k].total, n:porResponsable[k].n }; });
    responsables.sort(function(a,b){ return b.total - a.total; });
    var tendencia = Object.keys(porMes).sort().map(function(k){ return { periodo:k, monto:porMes[k] }; });
 
    return {
      ok: true,
      total_registros: items.length,
      monto_total: total,
      depositado: { monto: depMonto, n: nDep },
      pendiente:  { monto: penMonto, n: nPen },
      porCategoria: cat,
      porEmpresa: porEmpresa,
      porResponsable: responsables,
      tendencia: tendencia,
      generado: new Date().toISOString()
    };
  } catch (err) { return { ok: false, message: err.message }; }
}

 * ----------------------------------------------------------------------
 * REEMPLAZA al bloque de importación anterior (mismos nombres de función).
 * Borra el bloque viejo y pega este en su lugar.
 *
 * Novedad: _impNum() limpia "S/", espacios y separadores de miles antes
 * de convertir, así los montos guardados como TEXTO ahora sí se leen.
 *
 * PASOS:
 *   1) Reemplaza el bloque anterior por este (Ctrl+S).

 *   4) Revisa BS_Rendiciones: los montos deben aparecer como número.

var BS_REND_IMPORT_SHEETS = ['DETALLE 2025', 'DETALLE 2026'];
 
function _impFecha(v) {
  if (v instanceof Date && !isNaN(v.getTime())) return v;
  if (v === null || v === undefined || v === '') return '';
  var d = new Date(v);
  return isNaN(d.getTime()) ? '' : d;
}
 
function _impEstado(v) {
  var s = String(v == null ? '' : v).trim().toUpperCase();
  if (!s) return 'NO DEPOSITADO';
  if (s.indexOf('NO') >= 0 && s.indexOf('DEPOSIT') >= 0) return 'NO DEPOSITADO';
  if (s.indexOf('DEPOSIT') >= 0) return 'DEPOSITADO';
  return s; // valores raros (ej. '653') quedan tal cual para revisión manual
}
 
// Convierte a número aunque venga como texto: "S/ 1,234.56" -> 1234.56
function _impNum(v) {
  if (typeof v === 'number') return isNaN(v) ? 0 : v;
  if (v === null || v === undefined) return 0;
  var s = String(v).trim();
  if (!s) return 0;
  s = s.replace(/s\/\.?/i, '').replace(/\s/g, ''); // quita "S/" y espacios
  if (s.indexOf('.') === -1 && s.indexOf(',') > -1) {
    s = s.replace(/,/g, '.');   // coma usada como decimal
  } else {
    s = s.replace(/,/g, '');    // coma usada como miles
  }
  var n = parseFloat(s);
  return isNaN(n) ? 0 : n;
}

function bs_importarRendiciones() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var dest = ss.getSheetByName(BS_REND_SHEET);
  if (!dest) return 'ERROR: no existe BS_Rendiciones. Ejecuta bs_setupRendiciones primero.';
 
  var headers = dest.getRange(1, 1, 1, dest.getLastColumn()).getValues()[0].map(function(h){ return String(h).trim(); });
 
  var nextId = 1, existentes = {};
  var destLastRow = dest.getLastRow();
  if (destLastRow > 1) {
    var iId = headers.indexOf('id'), iRend = headers.indexOf('rendicion');
    var destData = dest.getRange(2, 1, destLastRow - 1, dest.getLastColumn()).getValues();
    destData.forEach(function(r){
      var idv = parseInt(r[iId]) || 0; if (idv >= nextId) nextId = idv + 1;
      var rd = String(r[iRend] || '').trim(); if (rd) existentes[rd] = true;
    });
  }
 
  var now = new Date();
  var nuevas = [], importadas = 0, omitidas = 0, sinHoja = [];
 
  BS_REND_IMPORT_SHEETS.forEach(function(nombre){
    var src = ss.getSheetByName(nombre);
    if (!src) { sinHoja.push(nombre); return; }
    var anioHoja = (String(nombre).match(/(\d{4})/) || [])[1] || '';
    var lr = src.getLastRow(); if (lr < 2) return;
    var lc = src.getLastColumn();
    var data = src.getRange(2, 1, lr - 1, lc).getValues();
 
    data.forEach(function(row){
      var fecha = _impFecha(row[0]);
      var rend  = String(row[2] || '').trim();
      if (!fecha && !rend) return;
      if (rend && existentes[rend]) { omitidas++; return; }
 
      var anio;
      if (fecha instanceof Date) anio = fecha.getFullYear();
      else if (anioHoja) anio = parseInt(anioHoja);
      else anio = now.getFullYear();
 
      var v = {
        id: nextId++,
        fecha: fecha,
        semana: parseInt(row[1]) || '',
        anio: anio,
        rendicion: rend,
        empresa: String(row[3] || '').trim(),
        monto_reclutamiento: _impNum(row[5]),
        monto_relaciones_laborales: _impNum(row[6]),
        monto_capacitaciones: _impNum(row[7]),
        monto_bienestar: _impNum(row[8]),
        monto_total: _impNum(row[4]),
        responsable: String(row[9] || '').trim(),
        estado: _impEstado(row[10]),
        fecha_deposito: _impFecha(row[11]),
        observaciones: '',
        created_by: 'IMPORT',
        created_at: now,
        updated_at: now
      };
      if (rend) existentes[rend] = true;
      nuevas.push(headers.map(function(h){ return (v[h] !== undefined) ? v[h] : ''; }));
      importadas++;
    });
  });
 
  if (nuevas.length) {
    dest.getRange(dest.getLastRow() + 1, 1, nuevas.length, headers.length).setValues(nuevas);
  }
 
  var msg = 'Importación rendiciones: ' + importadas + ' insertadas, ' + omitidas + ' omitidas (ya existían).';
  if (sinHoja.length) msg += '  ⚠ Hojas no encontradas: ' + sinHoja.join(', ') + '.';
  Logger.log(msg);
  return msg;
}

var BS_FIREBASE_URL = 'https://sistema-rl-verfrut-default-rtdb.firebaseio.com'; // <-- verifica que coincide con tu RTDB
 
// Escribe la señal. NUNCA rompe el guardado si Firebase falla.
function _notificarFirebase(accion) {
  try {
    var modulo = String(accion || '')
      .replace(/^bs_(crear|actualizar|eliminar)/, '')
      .toLowerCase() || 'general';
    var ahora = Date.now();
    var payload = {};
    payload['global'] = ahora;
    payload[modulo] = ahora;
    UrlFetchApp.fetch(BS_FIREBASE_URL + '/bienestar/senales.json', {
      method: 'patch',
      contentType: 'application/json',
      payload: JSON.stringify(payload),
      muteHttpExceptions: true
    });
  } catch (e) { /* silencioso: la señal nunca debe romper un registro */ }
}

// ---------- A) Migración de columnas + backfill ----------
function bs_migrarUsuariosPermisos() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName('Usuarios');
  if (!sh) return 'No existe la hoja Usuarios';
 
  var lastCol = sh.getLastColumn();
  var headers = sh.getRange(1, 1, 1, lastCol).getValues()[0].map(function(x){ return String(x).trim(); });
  var agregadas = [];
  BS_PERM_COLS_NUEVAS.forEach(function(h){
    if (headers.indexOf(h) === -1) {
      lastCol++;
      sh.getRange(1, lastCol).setValue(h)
        .setFontWeight('bold').setBackground('#0a2540').setFontColor('#ffffff').setHorizontalAlignment('center');
      headers.push(h); agregadas.push(h);
    }
  });
 
  // Backfill de usuarios existentes (sin quitar acceso a nadie)
  var lastRow = sh.getLastRow();
  var rellenados = 0;
  if (lastRow > 1) {
    var data = sh.getRange(1, 1, lastRow, sh.getLastColumn()).getValues();
    var H = data[0].map(function(x){ return String(x).trim(); });
    var iU = H.indexOf('usuario'), iR = H.indexOf('rol');
    var iRen = H.indexOf('mod_rendiciones'), iPan = H.indexOf('mod_panel'), iEst = H.indexOf('mod_estadisticas');
    for (var r = 1; r < lastRow; r++) {
      var u = String(data[r][iU] || '').trim().toLowerCase();
      if (!u) continue;
      var esAdmin = BS_ADMINS_CORE.indexOf(u) >= 0 || /admin|geren/i.test(String(data[r][iR] || ''));
      var cambios = false;
      function setSiVacio(colIdx, valor) {
        if (colIdx >= 0 && (data[r][colIdx] === '' || data[r][colIdx] === null || data[r][colIdx] === undefined)) {
          sh.getRange(r + 1, colIdx + 1).setValue(valor);
          cambios = true;
        }
      }
      setSiVacio(iRen, esAdmin ? true : false);
      setSiVacio(iPan, esAdmin ? true : true);   // todos conservan el panel que ya veían
      setSiVacio(iEst, esAdmin ? true : false);
      if (cambios) rellenados++;
    }
  }
  var msg = (agregadas.length ? 'Columnas agregadas: ' + agregadas.join(', ') : 'Columnas ya existían') +
            ' | Usuarios rellenados: ' + rellenados;
  Logger.log(msg);

  var data = lastRow > 1 ? sh.getRange(1, 1, lastRow, sh.getLastColumn()).getValues() : [headers];
  var filaDe = {};
  for (var r = 1; r < data.length; r++) {
    var u = String(data[r][ix('usuario')] || '').trim().toLowerCase();
    if (u) filaDe[u] = r + 1; // fila real en la hoja
  }
 
  var now = new Date(), creados = [], actualizados = [];
  USUARIOS.forEach(function(cfg){
    var flags = {
      mod_quejas: cfg.q, mod_accidentes: cfg.a, mod_hostigamiento: cfg.h, mod_subsidios: cfg.s,
      mod_rendiciones: cfg.r, mod_panel: cfg.p, mod_estadisticas: cfg.e
    };
    if (filaDe[cfg.u]) {
      // Existe: actualizar SOLO permisos (+ activo y updated_at). No toca clave ni rol.
      var fila = filaDe[cfg.u];
      Object.keys(flags).forEach(function(col){ sh.getRange(fila, ix(col) + 1).setValue(flags[col]); });
      if (ix('activo') >= 0) sh.getRange(fila, ix('activo') + 1).setValue(true);
      if (ix('updated_at') >= 0) sh.getRange(fila, ix('updated_at') + 1).setValue(now);
      actualizados.push(cfg.u);
    } else {
      // No existe: crear con clave inicial <usuario>2026 y rol 'usuario'
      var v = {
        usuario: cfg.u,
        password_hash: bs_hashPassword(cfg.u + '2026'),
        nombre: cfg.nombre,
        rol: 'usuario',
        email: '', telefono: '',
        mod_quejas: flags.mod_quejas, mod_accidentes: flags.mod_accidentes,
        mod_hostigamiento: flags.mod_hostigamiento, mod_subsidios: flags.mod_subsidios,
        mod_rendiciones: flags.mod_rendiciones, mod_panel: flags.mod_panel, mod_estadisticas: flags.mod_estadisticas,
        activo: true, created_at: now, updated_at: now
      };
      sh.appendRow(headers.map(function(h){ return (v[h] !== undefined) ? v[h] : ''; }));
      creados.push(cfg.u + ' (clave: ' + cfg.u + '2026)');
    }
  });
 
  var msg = 'Creados: ' + (creados.length ? creados.join(', ') : 'ninguno') +
            ' | Actualizados: ' + (actualizados.length ? actualizados.join(', ') : 'ninguno');
  Logger.log(msg);
  return msg;
}

 * BS-IMPORTAR-SUBSIDIOS · Backend Apps Script — Bienestar Social · Unifrutti
 * -------------------------------------------------------------------------
 * INSTALACIÓN (una sola vez):
 * 1. Abre script.google.com → tu proyecto de Bienestar Social.
 * 2. Crea un archivo nuevo (+ → Secuencia de comandos) llamado

 * 3. En tu función doPost(e), dentro del switch de acciones, agrega:
 *
 *      case 'bs_importarSubsidios':

 *        break;
 *
 * 4. Implementar → Administrar implementaciones → ✏️ → Nueva versión →
 *    Implementar. (NUNCA "Nueva implementación": conserva la URL fija.)
 *
 * QUÉ HACE:
 * Recibe { usuario, modo, filas:[{anio, mes, tipo, numero, ...}] } desde el
 * botón "📥 Importar Excel" del módulo Subsidios y hace UPSERT en la hoja
 * BS_Subsidios:
 *  - Si la fila coincide con un registro existente (mismo EXPEDIENTE, o
 *    mismo DNI + año + tipo + periodo) → ACTUALIZA solo los campos que
 *    vienen con valor (no pisa con vacíos).
 *  - Si no existe → AGREGA con id correlativo nuevo.
 *  - Nunca borra registros.
 * Lee los nombres de columna de la fila 1 de la hoja, así que funciona
 * aunque cambies el orden de las columnas.
 * ========================================================================= */
 
function bs_importarSubsidios(d) {
  try {
    if (!d || !d.filas || !d.filas.length) {
      return { ok: false, message: 'No llegaron filas para importar.' };
    }
    if (d.filas.length > 200) {
      return { ok: false, message: 'Máximo 200 filas por lote (el frontend envía lotes de 100).' };
    }
 
    var lock = LockService.getScriptLock();
    lock.waitLock(30000);
    try {
      var ss = SpreadsheetApp.getActive();
      var sh = ss.getSheetByName('BS_Subsidios');
      if (!sh) return { ok: false, message: 'No existe la hoja BS_Subsidios.' };
 
      var lastCol = sh.getLastColumn();
      var headers = sh.getRange(1, 1, 1, lastCol).getValues()[0];
      var idx = {};
      headers.forEach(function (h, i) { idx[String(h).trim()] = i; });
      if (idx['id'] === undefined) {
        return { ok: false, message: 'La hoja BS_Subsidios no tiene columna "id" en la fila 1.' };
      }
 
      var lastRow = sh.getLastRow();
      var data = lastRow > 1 ? sh.getRange(2, 1, lastRow - 1, lastCol).getValues() : [];
 
      // ---- clave de coincidencia (upsert) ----
      function claveDe(get) {
        var exp = String(get('expediente') || '').trim().toUpperCase();
        if (exp) return 'EXP:' + exp;
        return 'K:' + [
          String(get('numero') || '').trim(),
          String(get('anio') || '').trim(),
          String(get('tipo') || '').trim().toUpperCase(),
          String(get('periodo') || '').trim().toUpperCase()
        ].join('|');
      }
 
      var mapa = {};   // clave → índice de fila en `data`
      var maxId = 0;
      data.forEach(function (r, i) {
        var id = parseInt(r[idx['id']], 10);
        if (!isNaN(id) && id > maxId) maxId = id;
        var g = function (c) { return idx[c] === undefined ? '' : r[idx[c]]; };
        var k = claveDe(g);
        if (k !== 'K:|||' && mapa[k] === undefined) mapa[k] = i;
      });
 
      var creados = 0, actualizados = 0, errores = [];
      var ahora = new Date();
      var usuario = d.usuario || 'importacion';
      var soloAgregar = (d.modo === 'solo_agregar');
 
      d.filas.forEach(function (f, n) {
        try {
          var g = function (c) { return f[c] === undefined || f[c] === null ? '' : f[c]; };
          var k = claveDe(g);
 
          if (!soloAgregar && mapa[k] !== undefined) {
            // ---- ACTUALIZAR (solo campos con valor) ----
            var rowN = mapa[k] + 2; // +2: encabezado + base 1
            var rowVals = sh.getRange(rowN, 1, 1, lastCol).getValues()[0];
            for (var c in f) {
              if (c === 'id' || idx[c] === undefined) continue;
              var v = f[c];
              if (v === '' || v === null || v === undefined) continue;
              rowVals[idx[c]] = v;
            }
            if (idx['updated_at'] !== undefined) rowVals[idx['updated_at']] = ahora;
            sh.getRange(rowN, 1, 1, lastCol).setValues([rowVals]);
            actualizados++;
          } else {
            // ---- CREAR ----
            maxId++;
            var nueva = [];
            for (var i2 = 0; i2 < lastCol; i2++) nueva.push('');
            nueva[idx['id']] = maxId;
            for (var c2 in f) {
              if (c2 === 'id' || idx[c2] === undefined) continue;
              nueva[idx[c2]] = f[c2];
            }
            if (idx['created_by'] !== undefined) nueva[idx['created_by']] = usuario;
            if (idx['created_at'] !== undefined) nueva[idx['created_at']] = ahora;
            if (idx['updated_at'] !== undefined) nueva[idx['updated_at']] = ahora;
            sh.appendRow(nueva);
            // registrar en el mapa para no duplicar dentro del mismo lote
            data.push(nueva);
            if (k !== 'K:|||') mapa[k] = data.length - 1;
            creados++;
          }
        } catch (e) {
          errores.push('Fila ' + (n + 1) + ': ' + e.message);
        }
      });
 
      return { ok: true, creados: creados, actualizados: actualizados, errores: errores };
    } finally {
      lock.releaseLock();
    }
  } catch (e) {
    return { ok: false, message: 'Error al importar: ' + e.message };
  }
}

var BS_SEG = {
  TTL_HORAS: 12,            // duración del token (horas)
  EXENTAS: ['bs_login'],    // acciones que NO requieren token
  // Acciones que además del token exigen el permiso del módulo:
  PERMISOS: {
    quejas: ['bs_crearQueja', 'bs_listarQuejas', 'bs_actualizarQueja',
             'bs_listarSeguimientosQueja', 'bs_crearSeguimientoQueja'],
    accidentes: ['bs_crearAccidente', 'bs_listarAccidentes', 'bs_actualizarAccidente',
                 'bs_listarSeguimientos', 'bs_crearSeguimiento',
                 'bs_listarTodoHistorialAccidentes', 'bs_listarTodosSeguimientos'],
    hostigamiento: ['bs_crearHostigamiento', 'bs_listarHostigamiento', 'bs_actualizarHostigamiento'],
    subsidios: ['bs_crearSubsidio', 'bs_listarSubsidios', 'bs_actualizarSubsidio', 'bs_importarSubsidios']
  }
  // Cualquier otra acción bs_* (dashboard, alertas, documentos, rendiciones,
  // sugerencias, estadísticas, etc.) exige token válido, sin permiso extra.
};
 
/* ---------- EMITIR TOKEN (llamar desde bs_login) ---------- */
function bs_emitirToken(u) {
  var token = Utilities.getUuid() + Utilities.getUuid().slice(0, 8);
  var expira = Date.now() + BS_SEG.TTL_HORAS * 3600 * 1000;
  var data = {
    usuario: (u && u.usuario) || '',
    rol: (u && u.rol) || '',
    permisos: (u && u.permisos) || {},
    expira: expira
  };
  // Cache (rápido, máx 6 h) + hoja (respaldo hasta que expire)
  try {
    CacheService.getScriptCache().put('bs_tok_' + token, JSON.stringify(data),
      Math.min(21600, Math.floor(BS_SEG.TTL_HORAS * 3600)));
  } catch (e) {}
  var sh = _bsSheetTokens_();
  sh.appendRow([token, data.usuario, data.rol, JSON.stringify(data.permisos), new Date(), new Date(expira)]);
  _bsPurgarTokens_(sh);
  return token;
}
 
/* ---------- VALIDAR ACCESO (llamar desde doPost) ---------- */
function bs_validarAcceso(params) {
  try {
    var action = params && params.action;
    if (!action) return { ok: false, code: 400, message: 'Acción no especificada.' };
    if (BS_SEG.EXENTAS.indexOf(action) >= 0) return { ok: true };
 
    var token = params.token;
    if (!token) return _bsNoAutorizado_();
 
    var s = _bsBuscarToken_(String(token));
    if (!s || !s.expira || Date.now() > s.expira) return _bsNoAutorizado_();
 
    // Permiso por módulo (los admins pasan todo)
    var rol = String(s.rol || '').toLowerCase();
    if (rol.indexOf('admin') < 0) {
      for (var mod in BS_SEG.PERMISOS) {
        if (BS_SEG.PERMISOS[mod].indexOf(action) >= 0) {
          if (!s.permisos || !s.permisos[mod]) {
            return { ok: false, code: 403, message: 'No tienes permiso para el módulo "' + mod + '".' };
          }
          break;
        }
      }
    }
    params._sesion = s; // disponible para las acciones (quién es el usuario real)
    return { ok: true };
  } catch (e) {
    return { ok: false, code: 500, message: 'Error validando la sesión: ' + e.message };
  }
}
 
function _bsNoAutorizado_() {
  return { ok: false, code: 401, message: 'Sesión inválida o expirada. Vuelve a iniciar sesión.' };
}
 
/* ---------- helpers ---------- */
function _bsBuscarToken_(token) {
  // 1) cache
  try {
    var c = CacheService.getScriptCache().get('bs_tok_' + token);
    if (c) return JSON.parse(c);
  } catch (e) {}
  // 2) hoja
  var sh = _bsSheetTokens_();
  var last = sh.getLastRow();
  if (last < 2) return null;
  var vals = sh.getRange(2, 1, last - 1, 6).getValues();
  for (var i = vals.length - 1; i >= 0; i--) {
    if (String(vals[i][0]) === token) {
      var s = {
        usuario: vals[i][1],
        rol: vals[i][2],
        permisos: {},
        expira: vals[i][5] ? new Date(vals[i][5]).getTime() : 0
      };
      try { s.permisos = JSON.parse(vals[i][3] || '{}'); } catch (e) {}
      // re-cachear para las siguientes llamadas
      try {
        var restante = Math.floor((s.expira - Date.now()) / 1000);
        if (restante > 30) {
          CacheService.getScriptCache().put('bs_tok_' + token, JSON.stringify(s), Math.min(21600, restante));
        }
      } catch (e2) {}
      return s;
    }
  }
  return null;
}
 
function _bsSheetTokens_() {
  var ss = SpreadsheetApp.getActive();
  var sh = ss.getSheetByName('BS_Tokens');
  if (!sh) {
    sh = ss.insertSheet('BS_Tokens');
    sh.appendRow(['token', 'usuario', 'rol', 'permisos', 'creado', 'expira']);
    sh.getRange(1, 1, 1, 6).setFontWeight('bold');
    sh.hideSheet(); // hoja interna, no estorba
  }
  return sh;
}
 
// Limpia tokens vencidos cuando la hoja crece (se llama al emitir)
function _bsPurgarTokens_(sh) {
  try {
    var last = sh.getLastRow();
    if (last < 300) return;
    var vals = sh.getRange(2, 6, last - 1, 1).getValues();
    var ahora = Date.now();
    for (var i = vals.length - 1; i >= 0; i--) {
      var exp = vals[i][0] ? new Date(vals[i][0]).getTime() : 0;
      if (exp && exp < ahora) sh.deleteRow(i + 2);
    }
  } catch (e) {}
}
